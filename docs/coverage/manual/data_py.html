19def build_demo_case(case_index: int = 0) -> CaseData: 

20 """Create and return a case object by index (0-3).""" 

21 builders = [ 

22 _build_case_hit_and_run, 

23 _build_case_dead_mans_message, 

24 _build_case_ghost_employee, 

25 _build_case_silent_witness, 

26 ] 

27 if case_index < 0 or case_index >= len(builders): 

28 case_index = 0 

29 return builders[case_index]()
